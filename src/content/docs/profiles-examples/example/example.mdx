This minimal example shows the core v1.0 pattern: schema header, metadata, authority, roles, reading, block, section, ordo, proper, and teaching note.


```json
{
  "$schema": "https://openliturgy.org/schema/v1.0/corpus.schema.json",
  "ols_version": "1.0.0",
  "type": "corpus",
  "id": "eotc-kidase-demo-v1",
  "metadata": {
    "title": { "en": "EOTC Kidase Demo Corpus" }
  },
  "roles": [
    { "id": "role-deacon", "type": "deacon" }
  ],
  "blocks": [
    {
      "id": "block-kyrie",
      "type": "response",
      "elements": [
        {
          "utterance": { "roles": ["role-deacon"], "text": { "en": "Lord, have mercy" } }
        }
      ]
    }
  ]
}
```
